CREATE TABLE PokemonInfo
(
    PKMN_ID INT2 NOT NULL unique AUTO_INCREMENT, 
    PKMN_NAME VARCHAR(20) NOT null UNIQUE, 
    PRIMARY KEY(PKMN_ID)
);

CREATE TABLE Evolutions
(
    PKMN_ID int2 not null AUTO_INCREMENT,
    PKMN_NAME VARCHAR(20) not null UNIQUE, 
    EVOLUTION_ID INT2,  
    EVOLVING_ID INT2, 
    FOREIGN KEY(PKMN_NAME) REFERENCES PokemonInfo(PKMN_NAME),
    foreign key(PKMN_ID) references PokemonInfo(PKMN_ID)
);

CREATE TABLE RegionFound
(
    PKMN_REGION VARCHAR(50), 
    PKMN_ID INT2 NOT NULL AUTO_INCREMENT, 
    FOREIGN KEY(PKMN_ID) REFERENCES PokemonInfo(PKMN_ID)
);

CREATE TABLE TypeChart
(
    PKMN_TYPE1 VARCHAR(30) NOT NULL, 
    PKMN_TYPE2 VARCHAR(30), 
    PKMN_ID INT2 NOT NULL AUTO_INCREMENT, 
    FOREIGN KEY(PKMN_ID) REFERENCES PokemonInfo(PKMN_ID)
);

CREATE TABLE PokemonWeight
(
    PKMN_ID INT2 NOT NULL AUTO_INCREMENT, 
    PKMN_WEIGHT INT2 NOT NULL, 
    FOREIGN KEY(PKMN_ID) REFERENCES PokemonInfo(PKMN_ID)
); 

CREATE TABLE PokemonStats
(
    PKMN_ID INT2 NOT NULL AUTO_INCREMENT, 
    STATS_TOTAL INT2 NOT NULL, 
    STATS_HP INT2 NOT NULL, 
    STATS_ATTACK INT2 NOT NULL, 
    STATS_DEFENSE INT2 NOT NULL, 
    STATS_SP_ATTACK INT2 NOT NULL, 
    STATS_SP_DEFENSE INT2 NOT NULL, 
    STATS_SPEED INT2 NOT NULL, 
    FOREIGN KEY(PKMN_ID) REFERENCES PokemonInfo(PKMN_ID)
);
