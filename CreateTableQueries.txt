CREATE TABLE PokemonInfo(PKMN_ID INT2 NOT NULL AUTO_INCREMENT, PKMN_NAME VARCHAR(20) NOT NULL, PKMN_TYPE VARCHAR(20) NOT NULL, PRIMARY KEY(PKMN_ID));

CREATE TABLE PokemonStats(PKMN_ID INT2 NOT NULL AUTO_INCREMENT, STATS_TOTAL INT2 NOT NULL, STATS_HP INT2 NOT NULL, STATS_ATTACK INT2 NOT NULL, STATS_DEFENSE INT2 NOT NULL, STATS_SP_ATTACK INT2 NOT NULL, STATS_SP_DEFENSE INT2 NOT NULL, STATS_SPEED INT2 NOT NULL, PRIMARY KEY(STATS_TOTAL), FOREIGN KEY(PKMN_ID) REFERENCES PokemonInfo(PKMN_ID));

CREATE TABLE Evolutions(EVOLUTION_ID INT2 NOT NULL, PKMN_ID INT2 NOT NULL AUTO_INCREMENT, EVOLVING_ID INT2 NOT NULL, PRIMARY KEY(EVOLUTION_ID), FOREIGN KEY(PKMN_ID) REFERENCES PokemonInfo(PKMN_ID));

CREATE TABLE RegionFound(PKMN_REGION VARCHAR(50), PKMN_ID INT2 NOT NULL AUTO_INCREMENT, FOREIGN KEY(PKMN_ID) REFERENCES PokemonInfo(PKMN_ID));

CREATE TABLE TypeChart(PKMN_TYPE VARCHAR(30) NOT NULL, PKMN_ID INT2 NOT NULL AUTO_INCREMENT, FOREIGN KEY(PKMN_ID) REFERENCES PokemonInfo(PKMN_ID));
